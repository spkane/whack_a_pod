kubectl create deployment -f whack-a-pod-deployment.yaml

Deployment Configuration

apiVersion: apps/v1
kind: Deployment
metadata:
  labels:
    app: game
  name: api-deployment
  namespace: default
spec:
  progressDeadlineSeconds: 600
  replicas: 12
  revisionHistoryLimit: 10
  selector:
    matchLabels:
      app: api-deployment
  strategy:
    rollingUpdate:
      maxSurge: 25%
      maxUnavailable: 25%
    type: RollingUpdate
  template:
    metadata:
      labels:
        app: api-deployment
    spec:
      containers:
      - image: docker.io/spkane/whack-a-pod:api
        imagePullPolicy: IfNotPresent
        name: whack-a-pod
        ports:
        - containerPort: 8080
          protocol: TCP
        terminationMessagePath: /dev/termination-log
        terminationMessagePolicy: File
      dnsPolicy: ClusterFirst
      restartPolicy: Always
      schedulerName: default-scheduler
      terminationGracePeriodSeconds: 30

Service Configuration

apiVersion: v1
kind: Service
metadata:
  labels:
    app: api
  name: api
spec:
  externalTrafficPolicy: Cluster
  internalTrafficPolicy: Cluster
  ipFamilies:
  - IPv4
  ipFamilyPolicy: SingleStack
  ports:
  - nodePort: 30700
    port: 8080
    protocol: TCP
    targetPort: 8080
  selector:
    app: api-deployment
  sessionAffinity: None
  type: NodePort
